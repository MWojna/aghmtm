TASK(TA_ANA_INPUTS) {

    tWORD Luminosity;
    tWORD LUM;
    static tWORD Prev_LUM

    Luminosity = ReadLuminosity();
    LUM = ConversionDayNight(Luminosity);
    if(Prev_LUM != LUM) {
        Prev_LUM = LUM
        SetEvent(TA_CMD_AWNING, EV_CH_LUM);
    }
    TerminateTask();
}
