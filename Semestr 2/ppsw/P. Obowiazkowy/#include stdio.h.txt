#include <stdio.h>

void copyString(const char cSource[], char cDestination[]) {
		unsigned char cLoopCounter;
    for (cLoopCounter = 0; cSource[cLoopCounter] != NULL; cLoopCounter++) {
        cDestination[cLoopCounter] = cSource[cLoopCounter];
        if (cSource[cLoopCounter + 1] == NULL) { // Upewnienie sie, ze dodajemy znak NULL
            cDestination[cLoopCounter + 1] = NULL;
        }
    }
}





enum CompResult { DIFFERENT, EQUAL };

enum CompResult compareStrings(const char cStr1[], const char cStr2[]) {
		unsigned char cLoopCounter;
    for (cLoopCounter = 0; ; cLoopCounter++) {
        if (cStr1[cLoopCounter] != cStr2[cLoopCounter] || cStr1[cLoopCounter] == NULL) {
            return (cStr1[cLoopCounter] == cStr2[cLoopCounter]) ? EQUAL : DIFFERENT;
        }
    }
}





void vAppendString(char cDestinationStr[], const char cSourceStr[]) {
    // Znajdz koniec lancucha cDestinationStr
		unsigned char cSourceIndex;
    char cNullPos;
    for (cNullPos = 0; cDestinationStr[cNullPos] != NULL; cNullPos++);

    // Skopiuj cSourceStr do cDestinationStr zaczynajac od NULL-a
    for (cSourceIndex = 0; cSourceStr[cSourceIndex] != NULL; cSourceIndex++, cNullPos++) {
        cDestinationStr[cNullPos] = cSourceStr[cSourceIndex];
    }
}





void replaceChar(char cStr[], char cOldChar, char cNewChar) {
		unsigned char cLoopCounter;
    for (cLoopCounter = 0; cStr[cLoopCounter] != NULL; cLoopCounter++) {
        if (cStr[cLoopCounter] == cOldChar) {
            cStr[cLoopCounter] = cNewChar;
        }
    }
    printf("Modified String: %s\n", cStr);
}





int main() {
    const char cSource[] = "Testowy lancuch znakow";
    char cDestination[100]; // Pamietaj o odpowiednim rozmiarze

    copyString(cSource, cDestination);

    printf("Source: %s\n", cSource);
    printf("Destination: %s\n", cDestination);





 const char cStr1[] = "Test";
    const char cStr2[] = "Test";

    enum CompResult eResult = compareStrings(cStr1, cStr2);

    if (eResult == EQUAL) {
        printf("EQUAL\n");
    } else {
        printf("NOTEQUAL\n");
    }





    char cDestinationStr[100] = "Hello, ";
    const char cSourceStr[] = "World!";

    vAppendString(cDestinationStr, cSourceStr);

    printf("Resulting String: %s\n", cDestinationStr);





    char cStr[] = "Hello, World!";
    char cOldChar = 'o';
    char cNewChar = 'x';

    printf("Original String: %s\n", cStr);

    replaceChar(cStr, cOldChar, cNewChar);





    return 0;
}
